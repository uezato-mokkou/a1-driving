CREATE TABLE IF NOT EXISTS requests ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, car_id TEXT NOT NULL, customer_name TEXT, pickup TEXT, destination TEXT, note TEXT, status TEXT DEFAULT 'pending', created_at TIMESTAMPTZ DEFAULT NOW(), accepted_at TIMESTAMPTZ ); ALTER TABLE requests ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_all" ON requests FOR ALL USING (true) WITH CHECK (true); CREATE TABLE IF NOT EXISTS locations ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, name TEXT NOT NULL, lat DOUBLE PRECISION, lng DOUBLE PRECISION, category TEXT DEFAULT 'general', created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE locations ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_all" ON locations FOR ALL USING (true) WITH CHECK (true); CREATE TABLE IF NOT EXISTS staff ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, name TEXT NOT NULL UNIQUE, role TEXT NOT NULL DEFAULT '随行', pay_type TEXT NOT NULL DEFAULT 'hourly', hourly_rate INTEGER DEFAULT 900, is_active BOOLEAN DEFAULT true, created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE staff ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_all" ON staff FOR ALL USING (true) WITH CHECK (true); CREATE TABLE IF NOT EXISTS attendance ( id UUID DEFAULT gen_random_uuid() PRIMARY KEY, staff_id UUID, staff_name TEXT NOT NULL, work_date DATE NOT NULL, clock_in TIME, clock_out TIME, hours NUMERIC DEFAULT 0, total_sales INTEGER DEFAULT 0, commission INTEGER DEFAULT 0, hourly_pay INTEGER DEFAULT 0, total_pay INTEGER DEFAULT 0, paid BOOLEAN DEFAULT false, paid_date DATE, notes TEXT, created_at TIMESTAMPTZ DEFAULT NOW() ); ALTER TABLE attendance ENABLE ROW LEVEL SECURITY; CREATE POLICY "public_all" ON attendance FOR ALL USING (true) WITH CHECK (true);
